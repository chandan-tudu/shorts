Python Program To Add Two Matrices

Here's a Python program that adds two matrices:

def add_matrices(matrix1, matrix2):
    rows = len(matrix1)
    cols = len(matrix1[0])

    # Check if the matrices have the same dimensions
    if len(matrix2) != rows or len(matrix2[0]) != cols:
        return "Matrices must have the same dimensions for addition."

    result_matrix = [[0 for _ in range(cols)] for _ in range(rows)]

    for i in range(rows):
        for j in range(cols):
            result_matrix[i][j] = matrix1[i][j] + matrix2[i][j]

    return result_matrix

# Get input from the user for two matrices
def input_matrix():
    rows = int(input("Enter the number of rows: "))
    cols = int(input("Enter the number of columns: "))

    matrix = []
    for i in range(rows):
        row = []
        for j in range(cols):
            element = float(input(f"Enter the element at position ({i+1}, {j+1}): "))
            row.append(element)
        matrix.append(row)

    return matrix

print("Enter details for the first matrix:")
matrix1 = input_matrix()

print("\nEnter details for the second matrix:")
matrix2 = input_matrix()

# Add the matrices and display the result
result = add_matrices(matrix1, matrix2)
if isinstance(result, str):
    print(result)
else:
    print("\nThe sum of the matrices is:")
    for row in result:
        print(row)

This program prompts the user to enter the details for two matrices (number of rows and columns, and elements), then it adds the matrices and displays the result. Note that for matrix addition, the matrices must have the same dimensions.
